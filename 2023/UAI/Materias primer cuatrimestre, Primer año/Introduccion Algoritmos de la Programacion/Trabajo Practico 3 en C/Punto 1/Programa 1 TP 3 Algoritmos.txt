#include <stdio.h>
#include <stdlib.h>

int main(int argc, char* argv[])
{
	int Val, Cont, CantVeces;
	float Pro = 1, Sum = 0;
	//Se declara la cantidad de variables que quireas que el usuario ingrese.
	CantVeces = 25;

	for (Cont = 1; Cont <= CantVeces; Cont++)
	{
		printf("\nIngresa el valor entero\n"), Sum + 1;
		scanf_s("%d", &Val);
		Sum += Val;
	}

	Pro = Sum / CantVeces;

	printf("\nEl promedio de los %d valores es %f", CantVeces, Pro);
	return 0;
}
